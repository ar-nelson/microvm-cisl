  static class State {
    int op;
    State[] leaves;
    static final short INFINITE = Short.MAX_VALUE;
    short[] cost;
    short[] rule; 
 }
    
  public static void record(State p, int nt, int cost, int ruleno) {
    if (cost < p.cost[nt]) {
      p.cost[nt] = (short) cost;
      p.rule[nt] = (short) ruleno;
    }
  }

    State p = new State();
